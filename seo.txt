export async function generateMetadata({ params }) {

  const college = await getCollege(params.slug);

    openGraph: { title: ..., images: [college.banner_url] },

    alternates: { canonical: `/colleges/${params.stream}/${params.slug}` }

  }

}

3.2 sitemap.ts — Complete Implementation

// app/sitemap.ts

import { MetadataRoute } from "next"

import { createClient } from "@/lib/supabase/server"



export default async function sitemap(): Promise<MetadataRoute.Sitemap> {

  const supabase = createClient()

  const base = "https://edumadras.com"



  // Fetch all colleges

  const { data: colleges } = await supabase

    .from("colleges").select("slug, stream, updated_at")



  // Fetch all courses

  const { data: courses } = await supabase

    .from("master_courses").select("slug, stream")



  const staticPages = [

    { url: base, changeFrequency: "weekly", priority: 1.0 },

    { url: `${base}/colleges`, changeFrequency: "daily", priority: 0.9 },

    { url: `${base}/counselors`, changeFrequency: "weekly", priority: 0.8 },

    ["engineering","medical","management","law","design"].map(s => ({

      url: `${base}/colleges/${s}`, changeFrequency: "daily", priority: 0.9

    }))

  ].flat()



  const collegePages = colleges.map(c => ({

    url: `${base}/colleges/${c.stream.toLowerCase()}/${c.slug}`,

    lastModified: new Date(c.updated_at),

    changeFrequency: "weekly", priority: 0.8

  }))



  return [...staticPages, ...collegePages, ...coursePages]

}

4.3 Internal Linking Strategy — The PageRank Flow Map

Internal links pass authority between pages. Every link from a high-authority page to a lower-authority page strengthens the target. Build this deliberately.



The Authority Flow Hierarchy

LEVEL 1 (Highest Authority): Homepage → receives most external links

LEVEL 2: Stream listing pages (/colleges/engineering) → linked from homepage and all college cards

LEVEL 3: College detail pages → linked from stream pages, compare pages, exam pages

LEVEL 4: Course/exam/city pages → linked from college detail pages and listings



RULE: Every page must link UP to its parent in the hierarchy.

RULE: Every page must link SIDEWAYS to 3-5 related pages (similar colleges).

RULE: Stream badge on college card = link to /colleges/[stream] (do NOT forget this).

RULE: Every city mention on a page = link to /cities/[city] automatically.

RULE: Every exam mention = link to /entrance-exams/[exam] automatically.

RULE: Compare page links back to BOTH college detail pages in the breadcrumb.
